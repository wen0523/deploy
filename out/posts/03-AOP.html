    @Around("@annotation(cache)")
    public Object cacheData(ProceedingJoinPoint joinPoint, Cache cache) throws Throwable {
        // ç”Ÿæˆç¼“å­˜key

        // å°è¯•ä»ç¼“å­˜è·å–
        Object cachedResult = redisTemplate.opsForValue().get(cacheKey);
        if (cachedResult != null) {
            log.info("ä»ç¼“å­˜è·å–æ•°æ®: {}", cacheKey);
            return cachedResult;

        // å­˜å…¥ç¼“å­˜

                    cache.expire(), TimeUnit.SECONDS);

        }

    private String generateCacheKey(ProceedingJoinPoint joinPoint, String keyPrefix) {
        String className = joinPoint.getTarget().getClass().getSimpleName();
        String methodName = joinPoint.getSignature().getName();
        Object[] args = joinPoint.getArgs();

        if (keyPrefix.isEmpty()) {

                    Arrays.toString(args));

    }

@Aspect
@Component
@Slf4j
public class RateLimitAspect {

    @Autowired
    private RedisTemplate<String, String> redisTemplate;

    @Before("@annotation(rateLimit)")
    public void checkRateLimit(JoinPoint joinPoint, RateLimit rateLimit) {
        String key = generateRateLimitKey(joinPoint);

        // è·å–å½“å‰è®¿é—®æ¬¡æ•°
        String countStr = redisTemplate.opsForValue().get(key);
        int count = countStr == null ? 0 : Integer.parseInt(countStr);

        if (count >= rateLimit.limit()) {
            log.warn("æ¥å£é™æµè§¦å‘: {}, å½“å‰è®¿é—®æ¬¡æ•°: {}", key, count);
            throw new RateLimitException("è®¿é—®è¿‡äºé¢‘ç¹ï¼Œè¯·ç¨åå†è¯•");
        }

        // å¢åŠ è®¿é—®æ¬¡æ•°
        if (count == 0) {
            redisTemplate.opsForValue().set(key, "1",
                    rateLimit.period(), TimeUnit.SECONDS);
        } else {
            redisTemplate.opsForValue().increment(key);
        }
    }

    private String generateRateLimitKey(JoinPoint joinPoint) {
        String className = joinPoint.getTarget().getClass().getSimpleName();
        String methodName = joinPoint.getSignature().getName();
        String userIp = getCurrentUserIp();

        return String.format("rate_limit:%s:%s:%s", className, methodName, userIp);
    }

    private String getCurrentUserIp() {
        ServletRequestAttributes attributes =
            (ServletRequestAttributes) RequestContextHolder.getRequestAttributes();
        if (attributes != null) {
            return attributes.getRequest().getRemoteAddr();
        }
        return "unknown";
    }
}
